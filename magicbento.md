# INSTALL

```bash
npx shadcn@latest add @react-bits/MagicBento-TS-TW
```

# USAGE

```typescript
import MagicBento from './MagicBento'

<MagicBento 
  textAutoHide={true}
  enableStars
  enableSpotlight
  enableBorderGlow={true}
  enableTilt={false}
  enableMagnetism={false}
  clickEffect
  spotlightRadius={400}
  particleCount={12}
  glowColor="132, 0, 255"
  disableAnimations={false}
/>

```
